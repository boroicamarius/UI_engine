    • Description
    • This is the base class of all UI derived elements
      Note: you can use this for rendering but it wont render anything
    • Use
    • This class was meant to be used for derived classes
    • Example 
        #include "SDL.h"
  #include "SDL_Image.h"
  #include "SDL_ttf.h"
  #include "UI_engine.h"

  using namespace UI_engine;

  int main(int argc, char* argv[]) {
    UI_window window("exampleWindow",
      SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,   
      400, 400, SDL_WINDOW_SHOWN, SDL_RENDERER_ACCELERATED,
      144.0);
      
    UI example();
    
    window.add(&example);
    processEvents(&window);
    return 0;
  }

    • Description
    • Something that creates a filled/not filled rectangle based on the coordinates and dimensions given
      Note: Since this is a derived class of UI check the UI documentation
    • Example 
        #include "SDL.h"
  #include "SDL_Image.h"
  #include "SDL_ttf.h"

      SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,

      144.0);
      
    UI_frame frame_example(0.0f,0.0f,35.0f,35.0f,1.0f,{0,0,0,255});
    frame_example.filled();

    window.add(&frame_example);
    processEvents(&window);
    return 0;
  }

    • Description
    • A window used to render UI_elements(it can do a lot of things(no joke))
      Note: you can make only a window and it runs in a single thread so think about it
    • Example 
        #include "SDL.h"
  #include "SDL_Image.h"
  #include "SDL_ttf.h"
  #include "UI_engine.h"

  using namespace UI_engine;

  int main(int argc, char* argv[]) {
    UI_window window("exampleWindow",
      SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,   
      400, 400, SDL_WINDOW_SHOWN, SDL_RENDERER_ACCELERATED,
      144.0);

    processEvents(&window);//remember to use this function to actually make your window render things
    return 0;
  }

    • Description
    • Stores UI_objects and groups them into a single UI_object for a better control
      Note: Since this is a derived class of UI check the UI documentation
    • Example 
        #include "SDL.h"
  #include "SDL_Image.h"
  #include "SDL_ttf.h"
  #include "UI_engine.h"

  using namespace UI_engine;

  int main(int argc, char* argv[]) {
    UI_window window("exampleWindow",
      SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,   
      400, 400, SDL_WINDOW_SHOWN, SDL_RENDERER_ACCELERATED,
      144.0);

    UI_collection collection_example(35.0f,60.0f,1.0f);

    UI_text some_text(0.0f,0.0f,1.0f,"Here is the text");

    collection_example.add(&some_text);
    window.add(&collection_example);

    processEvents(&window);
    return 0;
  }
